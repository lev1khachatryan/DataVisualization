## Grid style (cont'd) -------------------------- 

```
import dash
import dash_core_components as dcc
import dash_html_components as html

import plotly.graph_objs as go
x_values = ['HTML/CSS','Python','SQL','Other']
y_values = [15,50,15,20]
trace = go.Bar(x=x_values, y=y_values)
data = [trace]
my_figure = dict(data=data)

app = dash.Dash()
app.css.append_css({'external_url': 'https://codepen.io/chriddyp/pen/bWLwgP.css'})

app.layout = html.Div([

	html.Div([

	    html.Label('Dropdown'),
    	dcc.Dropdown(
        options=[
            {'label': 'Facebook', 'value': 'www.fb.com'},
            {'label': 'Twitter', 'value': 'www.twitter.com'}
        ],
        value='Facebook')

	], className='two columns'),


	html.Div([

		html.H1("My figure"),
		dcc.Graph(id='figure',figure=my_figure)

	], className='six columns')
	
], className='row')

if __name__ == '__main__':
    app.run_server(debug=True)
```

### Reactive Programming
## Reactive text input -------------------------- 

```
import dash
import dash_core_components as dcc
import dash_html_components as html
from dash.dependencies import Input, Output

app = dash.Dash()

app.layout = html.Div([
    dcc.Input(id='text_in', value='Name Surname', type='text'),
    html.Div(id='text_out')
])


@app.callback(
    Output(component_id='text_out', component_property='children'),
    [Input(component_id='text_in', component_property='value')]
)
def update_text_output(input_value):
    return "Your name is " + input_value


if __name__ == '__main__':
    app.run_server()
```
